// w ordenado mayor→menor | W: objetivo
function coin_change(w[1..N], W)
  x[1..N] = [0..0]
  total = 0
  while total < W do
    i = best_coin(w, W − total)
    if i == −1 then return NULL
    x[i] += 1
    total = total + w[i]
  end_while
  return x
