// Objetos ordenados por mayor ratio v/w
function backpack(w[1..N], v[1..N], W)
  x[1..N] = [0..0]
  peso = 0   valor = 0   i = 1
  while (i ≤ N) and (peso < W) do
    if (peso + w[i]) ≤ W then
      x[i] = 1
      peso += w[i]   valor += v[i]
    else
      x[i] = (W − peso) / w[i]
      peso = W   valor += x[i]·v[i]
    end_if
    i += 1
  end_while
  return x, peso, valor
